  --train_csv ~/github/EMULaToR/data/processed/baselines/CatPred/enzyme_sequence_splits/threshold_0.09/train.parquet \
  --val_csv ~/github/EMULaToR/data/processed/baselines/CatPred/enzyme_sequence_splits/threshold_0.09/val.parquet \
  --test_csv ~/github/EMULaToR/data/processed/baselines/CatPred/enzyme_sequence_splits/threshold_0.09/test.parquet \
  --out_dir ~/github/EMULaToR/data/processed/baselines/CatPred/manual_runs/enz_seq_0_09/seed_42 \
  --sequence_col sequence \
  --uniprot_id_col pdbs \
  --smiles_columns smiles \
  --target_columns log10_value \
  --seed 42 \
  --epochs 30 \
  --batch_size 16 \
  --ensemble_size 10 \
  --num_workers 8 \
  --device cuda:0 \
  --mixed_precision auto \
  --optimizer_fused auto \
  --lr_scheduler cosine_warmup \
  --val_every_n_epochs 2 \
  --final_epoch_metrics_only \
  --early_stopping_patience 4 \
  --metric rmse \
  --ram_budget_gb 90 \
  --cache_dir ~/github/EMULaToR/data/processed/baselines/CatPred/embeddings \
  --require_cached_esm \

  --loss_function mve \
